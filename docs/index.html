prioritni_tech_googlesoft <- csi_recoded %>%
  filter(text_recoded %in% c("G Hangout/Meet/Classroom", "Microsoft Teams")) %>%
  mutate(pruzkum_grouped = 
           case_when(text_recoded == "G Hangout/Meet/Classroom" ~ "Google",
                     text_recoded == "Microsoft Teams" ~ "MS",
                     TRUE ~ "jiné/žádné"),
         g_in_survey = text_recoded == "G Hangout/Meet/Classroom",
         ms_in_survey = text_recoded == "Microsoft Teams") %>%
  select(red_izo, ends_with("in_survey"), pruzkum_grouped, text_recoded,
         text_detail)
